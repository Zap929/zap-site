    const PAIRS = [

    const BINANCE_PRICE_URL = (symbol) =>

      \`https://data-api.binance.vision/api/v3/ticker/price?symbol=\${symbol}\`;

    const BINANCE_24HR_URL = (symbol) =>

      \`https://data-api.binance.vision/api/v3/ticker/24hr?symbol=\${symbol}\`;

 

    // ------------------------------- Utilities ----------------------------------

    function useInterval(callback, delay) {

      const savedRef = useRef(() => {});

      useEffect(() => { savedRef.current = callback; }, [callback]);

      useEffect(() => {

        if (delay == null) return;

        const id = setInterval(() => savedRef.current(), delay);

        return () => clearInterval(id);

      }, [delay]);

    }

 

    function classNames(...xs) {

      return xs.filter(Boolean).join(" ");

    }

 

    function formatPrice(n) {

      if (!isFinite(n)) return "-";

      if (n >= 1000) return n.toLocaleString(undefined, { maximumFractionDigits: 2 });

      if (n >= 1) return n.toLocaleString(undefined, { maximumFractionDigits: 4 });

      return n.toLocaleString(undefined, { maximumFractionDigits: 6 });

    }

 

    // -------------------------------- Components --------------------------------

    function TickerChip({ label, price, prevPrice, changePct, volume }) {

      const numeric = Number(price);

      const prev = Number(prevPrice);

      const moved = !isNaN(numeric) && !isNaN(prev) ? numeric - prev : 0;

      const direction = moved === 0 ? "flat" : moved > 0 ? "up" : "down";

      const pretty = useMemo(() => formatPrice(numeric), [numeric]);

      const pct = typeof changePct === "number" && isFinite(changePct) ? changePct : null;

      const vol = typeof volume === "number" && isFinite(volume) ? volume : null;

 

      return (

        <div

          className={classNames(

            "flex items-center gap-2 rounded-full px-3 py-1 text-sm",

            direction === "up" && "bg-emerald-950/50 text-emerald-300 ring-1 ring-emerald-600/30",

            direction === "down" && "bg-rose-950/50 text-rose-300 ring-1 ring-rose-600/30",

            direction === "flat" && "bg-zinc-800/50 text-zinc-200 ring-1 ring-zinc-600/30"

          )}

          title={\`\${label} price \${direction === "flat" ? "unchanged" : direction === "up" ? "up" : "down"} vs last update\`}

        >

          <span className="font-semibold tracking-wide">{label}</span>

          <span className="opacity-80">${pretty}</span>

          {pct != null && (

            <span

              className={classNames(

                "rounded-full px-2 py-0.5 text-[11px]",

                pct > 0

                  ? "bg-emerald-900/60 text-emerald-300"

                  : pct < 0

                  ? "bg-rose-900/60 text-rose-300"

                  : "bg-zinc-800/80 text-zinc-300"

              )}

            >

              {pct > 0 ? "+" : ""}

              {pct.toFixed(2)}%

            </span>

          )}

          {vol != null && (

            <span className="hidden md:inline text-[11px] text-zinc-400">

              Vol {Intl.NumberFormat(undefined, { notation: "compact" }).format(vol)}

            </span>

          )}

          {direction === "up" && <span aria-hidden>▲</span>}

          {direction === "down" && <span aria-hidden>▼</span>}

          {direction === "flat" && <span aria-hidden>○</span>}

        </div>

      );

    }

 

    function CryptoTicker({ pairs = PAIRS, refreshMs = 15000 }) {

      const [prices, setPrices] = useState({});

      const [prevPrices, setPrevPrices] = useState({});

      const [changes, setChanges] = useState({}); // 24h %

      const [volumes, setVolumes] = useState({}); // 24h volume

      const [stamp, setStamp] = useState(null);

      const mounted = useRef(false);

 

      const fetchAll = async () => {

        try {

          const results = await Promise.all(

            pairs.map(async (p) => {

              const [priceRes, statRes] = await Promise.all([

                fetch(BINANCE_PRICE_URL(p.symbol), { cache: "no-store" }),

                fetch(BINANCE_24HR_URL(p.symbol), { cache: "no-store" }),

              ]);

              if (!priceRes.ok) throw new Error(\`\${p.symbol} price \${priceRes.status}\`);

              if (!statRes.ok) throw new Error(\`\${p.symbol} 24hr \${statRes.status}\`);

              const priceJson = await priceRes.json();

              const statJson = await statRes.json();

              return {

                symbol: p.symbol,

                price: Number(priceJson.price),

                changePct: Number(statJson.priceChangePercent),

                volume: Number(statJson.volume),

              };

            })

          );

          setPrevPrices((curr) => ({ ...curr }));

          setPrices(Object.fromEntries(results.map((r) => [r.symbol, r.price])));

          setChanges(Object.fromEntries(results.map((r) => [r.symbol, r.changePct])));

          setVolumes(Object.fromEntries(results.map((r) => [r.symbol, r.volume])));

          setStamp(new Date());

        } catch (err) {

          console.error("Price/24h fetch error:", err);

        }

      };

 

      useEffect(() => {

        if (!mounted.current) {

          mounted.current = true;

          fetchAll();

        }

      }, []);

 

      useInterval(fetchAll, refreshMs ?? 15000);

 

      const chips = pairs.map((p) => (

        <TickerChip

          key={p.symbol}

          label={p.label}

          price={prices[p.symbol] ?? 0}

          prevPrice={prevPrices[p.symbol] ?? 0}

          changePct={changes[p.symbol]}

          volume={volumes[p.symbol]}

        />

      ));

 

      const marqueeContent = (

        <div className="flex gap-3 pr-6">

          {chips}{chips}{chips}

        </div>

      );

 

      return (

        <div className="sticky top-0 z-50 w-full bg-zinc-900/95 backdrop-blur supports-[backdrop-filter]:bg-zinc-900/70 border-b border-zinc-800">

          <div className="mx-auto max-w-7xl px-3 sm:px-6">

            <div className="flex items-center justify-between py-2">

              <div className="flex items-center gap-3">

                <span className="text-xs font-semibold uppercase tracking-widest text-zinc-400">Live Crypto</span>

                {stamp && (

                  <time className="text-[11px] text-zinc-500" dateTime={stamp.toISOString()}>

                    Updated {stamp.toLocaleTimeString()}

                  </time>

                )}

              </div>

              <div className="text-[11px] text-zinc-500">Data via Binance 24hr & price endpoints</div>

            </div>

          </div>

          <div className="relative overflow-hidden border-t border-zinc-800">

            <div className="animate-marquee whitespace-nowrap will-change-transform">{marqueeContent}</div>

          </div>

        </div>

      );

    }
